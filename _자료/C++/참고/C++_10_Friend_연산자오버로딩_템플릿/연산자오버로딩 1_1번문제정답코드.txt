#include <iostream>
using namespace std;
class Point
{
	int x, y;
public:
	Point() {}
	Point(int a, int b) { x = a; y = b; }
	void Print() { cout << "x = " << x << ", y = " << y << "\n"; }
	friend Point operator / (Point tmp, Point tmp2);
};
Point operator / (Point tmp, Point tmp2)
{
