2. 세 과목의 성적을 입력 받아 합계와 평균을 구하고 평균이 90이상이면 “A”, 
80이상이면 “B”, 70 이상이면 “C”, 60 이상이면 “ D”, 60미만이면 “F”를 출력하시오.(if문)

int a, b,c,sum = 0;
cin >> a>>b>>c;
sum = a+b+c;
if(sum >= 90)
  cout << "A";
else if(sum >= 80)
  cout << "B";
else if(sum >= 70)
  cout << "C";
else if(sum >= 60)
  cout << "D";
else
  cout << "F"

3.1~1000까지의 합을 구하라.단 3의 배수는 제외하고 3의 배수이면서
  5의 배수는 제외하지 않는 조건으로 구하라.(for문)
  
  int sum = 0;
  for(int i = 1; i<=1000; i++)
  {
    if(i % 3 == 0 && i % 5 != 0)
      continue;
      sum += i;
  }

5. updown 게임을 완성하시오.

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void)
{
	int a;

	int number, p, count, best = 100, bestcount = 0;
	srand((unsigned)time(NULL));

	while (1)
	{
		printf("===업 & 다운 게임===\n");
		printf("	1. 게임시작\n");
		printf("	2. 게임끝내기\n");
		count = 0;
		scanf("%d", &a);
		number = rand() % 100;
		switch (a)
		{
		case 1:
			printf("게임시작\n");
			while (1)
			{

				scanf("%d", &p);
				bestcount++;
				count++;
				if (p == number)
				{
					printf("정답\n");
					if (count < best)
						best = count;
					break;
				}
				else if (p > number)
					printf("down\n");
				else
					printf("up\n");

			}
		}
		if (a == 2)
			break;
	}
}
